package com.user.controller; import java.io.IOException; import java.sql.Connection; import java.sql.DriverManager; import java.sql.PreparedStatement; import javax.servlet.ServletException; import javax.servlet.annotation.WebServlet; import javax.servlet.http.HttpServlet; import javax.servlet.http.HttpServletRequest; import javax.servlet.http.HttpServletResponse; public class RegisterServlet extends HttpServlet { @Override protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException { String uname = request.getParameter("username"); String fname = request.getParameter("fullname"); String uemail = request.getParameter("email"); String uphone = request.getParameter("phoneNumber"); String urole = request.getParameter("role"); String upass = request.getParameter("password"); String uconfirmPass = request.getParameter("confirmPassword"); if (upass == null || !upass.equals(uconfirmPass)) { request.setAttribute("errorMessage", "Password doesn't match!"); request.getRequestDispatcher("register.jsp").forward(request, response); return; } String url = "jdbc:derby://localhost:1527/LostAndFound"; String user = "app"; String password = "app"; try { Class.forName("org.apache.derby.jdbc.ClientDriver"); Connection conn = DriverManager.getConnection(url, user, password); String registerQuery = "INSERT INTO \"user\" (username, fullname, email, phone, role, password) VALUES (?, ?, ?, ?, ?, ?)"; PreparedStatement ps = conn.prepareStatement(registerQuery); ps.setString(1, uname); ps.setString(2, fname); ps.setString(3, uemail); ps.setString(4, uphone); ps.setString(5, urole); ps.setString(6, upass); int result = ps.executeUpdate(); ps.close(); conn.close(); if (result > 0) { response.sendRedirect("login.jsp"); } else { request.setAttribute("errorMessage", "Pendaftaran gagal. Sila cuba lagi."); request.getRequestDispatcher("register.jsp").forward(request, response); } } catch (Exception e) { e.printStackTrace(); request.setAttribute("errorMessage", "Database Error: " + e.getMessage()); request.getRequestDispatcher("register.jsp").forward(request, response); } } }
