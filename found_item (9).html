package com.user.controller; import java.io.IOException; import java.io.InputStream; import java.sql.Connection; import java.sql.DriverManager; import java.sql.PreparedStatement; import javax.servlet.ServletException; import javax.servlet.annotation.MultipartConfig; import javax.servlet.http.HttpServlet; import javax.servlet.http.HttpServletRequest; import javax.servlet.http.HttpServletResponse; import javax.servlet.http.Part; import java.io.BufferedReader; import java.io.InputStreamReader; import java.util.stream.Collectors; @MultipartConfig // PENTING: Mesti ada untuk proses file upload public class LostItemServlet extends HttpServlet { // Helper method untuk baca teks dari Part private String getValue(Part part) throws IOException { BufferedReader reader = new BufferedReader(new InputStreamReader(part.getInputStream(), "UTF-8")); return reader.lines().collect(Collectors.joining("\n")); } @Override protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException { // Guna request.getPart() untuk semua input apabila guna enctype="multipart/form-data" String itemName = getValue(request.getPart("itemName")); String category = getValue(request.getPart("category")); String location = getValue(request.getPart("location")); String dateLost = getValue(request.getPart("dateLost")); String description = getValue(request.getPart("description")); String contact = getValue(request.getPart("contact")); // SQL (Pastikan anda telah ALTER TABLE untuk tambah IMAGE dan STATUS) String sql = "INSERT INTO APP.LOST_ITEMS (ITEM_NAME, CATEGORY, LOCATION, DATE_LOST, DESCRIPTION, CONTACT, STATUS) VALUES (?, ?, ?, ?, ?, ?, ?)"; try { Class.forName("org.apache.derby.jdbc.ClientDriver"); Connection conn = DriverManager.getConnection("jdbc:derby://localhost:1527/LostAndFound", "app", "app"); PreparedStatement ps = conn.prepareStatement(sql); ps.setString(1, itemName); ps.setString(2, category); ps.setString(3, location); ps.setString(4, dateLost); ps.setString(5, description); ps.setString(6, contact); ps.setString(7, "ACTIVE"); // Default status ps.executeUpdate(); response.sendRedirect("dashboard.jsp?msg=success"); conn.close(); } catch (Exception e) { e.printStackTrace(); response.getWriter().println("Error: " + e.getMessage()); } } }
